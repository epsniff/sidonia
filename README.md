--------


#### Env Setup

```sh
# install minikube
curl -Lo minikube https://storage.googleapis.com/minikube/releases/v0.13.0/minikube-linux-amd64 && \
 chmod +x minikube && \
 sudo mv minikube /usr/local/bin/


# install kubectl
curl -Lo kubectl https://storage.googleapis.com/kubernetes-release/release/v1.4.5/bin/linux/amd64/kubectl && \
 chmod +x kubectl && \
 sudo mv kubectl /usr/local/bin/

```
